134 Vec<float> RRTPlanner::computeNodeNew(const Vec<float> &randNode, const Vec<float> &nearestNode) {
135  if (randNode.getDist(nearestNode) < this->m_stepSize)
136  return randNode;
137 
138  // p = a + k * (b-a)
139  // ||u|| = ||b - a||
140  // k = stepSize / ||u||
141  Vec<float> u = randNode - nearestNode;
142  u *= this->m_stepSize / u.norm();
143  u += nearestNode;
144  return u;
145 }
